\section*{Parte A -- Lógica Proposicional Booleana}

\Qitem{ \Qq{Julgue a afirmativa a seguir como \textbf{verdadeira} ou \textbf{falsa}:
    ``Afirmar que uma proposição $p$ é, ao mesmo tempo, verdadeira e falsa, violaria o Princípio do Terceiro Excluído''.
} }
% F. Feedback: O princípio violado seria o Princípio da Não-Contradição.

\Qitem{ \Qq{Na canção Maricotinha, Dorival Caymmi afirma que ``Se fizer bom tempo amanhã eu vou''. Assumindo que esta proposição é verdadeira, assinale a única situação impossível: }
\begin{Qlist}
\item Fez bom tempo e ele foi
\item Fez bom tempo e ele não foi
\item Não fez bom tempo e ele foi
\item Não fez bom tempo e ele não foi
\end{Qlist}
}
% B. Feedback: A condicional só é falsa caso o antecedente seja verdadeiro e o consequente falso.

\Qitem{ \Qq{Qual é o menor número natural $x$ que torna a sentença aberta ``$x/2 > \pi$'' verdadeira? } }

% Resposta: x = 7

% Feedback: $x$ tem que ser maior do que $2\pi$ e  $2\pi < 7$.
